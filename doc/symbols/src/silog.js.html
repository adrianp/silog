 29 /**
 30  * Revealing module for silog; exposes the required public members.
 31  *
 32  * @return {Object.<string,(function|Object)>} the exported public members.
 33  * @namespace
 34  */
 35 var silog = function() {
 36 
 37     /**
 38      * The logging levels used by silog; an order relation is established
 39      * between them, with ASSERT (7) being the highest and VERBOSE (2) the
 40      * lowest. If silog's level is set to X, any message with the level Y < X
 41      * will not be logged.
 42      * @enum {Array.<number, string>}
 43      * @const
 44      * @memberOf silog
 45      * @public
 46      */
 47     var LEVEL = {ASSERT: [7, 'ASSERT'],
 48                  ERROR: [6, 'ERROR'],
 49                  WARN: [5, 'WARN'],
 50                  INFO: [4, 'INFO'],
 51                  DEBUG: [3, 'DEBUG'],
 52                  VERBOSE: [2, 'VERBOSE'] };
 53 
 54     /**
 55      * The date/ time format used in the logged messages.
 56      * @enum {number}
 57      * @const
 58      * @memberOf silog
 59      * @public
 60      */
 61     var DT_FORMAT = {DATE_TIME: 0,
 62                      DATE: 1,
 63                      TIME: 2 };
 64 
 65     /**
 66      * The tag used for logging internal messages.
 67      * @type {String}
 68      * @private
 69      * @const
 70      */
 71     var LOCAL_TAG = 'silog';
 72 
 73     /**
 74      * Checks if the given parameter is a valid silog logging level i.e. is an
 75      * array of size 2, with the first element a number and the second one a
 76      * string. Moreover, if the second parameter evaluates to true, the
 77      * existence of the array in silog.LEVEL will be checked.
 78      * @param  {*}       level  the object to check.
 79      * @param  {boolean} exists if true, the existence of the object in
 80      *                          silog.LEVEL will be checked.
 81      * @return {boolean}        true if the object is a valid logging level,
 82      *                          false otherwise.
 83      * @private
 84      */
 85     function checkLevel(level, exists) {
 86         if (!Array.isArray(level) ||
 87             level.length !== 2 ||
 88             typeof level[0] !== 'number' ||
 89             typeof level[1] !== 'string') {
 90             return false;
 91         }
 92         if (exists) {
 93             for (var key in LEVEL) {
 94                 if (LEVEL.hasOwnProperty(key)) {
 95                     if (LEVEL[key][0] === level[0] &&
 96                        LEVEL[key][1] === level[1]) {
 97                         return true;
 98                     }
 99                 }
100             }
101             return false; // level not in silog.LEVEL
102         }
103         return true; // level valid, existence in silog.LEVEL unknown
104     }
105 
106     /**
107      * Adds a leading zero to numbers lower than 10 (e.g., 6 becomes 06, 11
108      * remains as is).
109      * @param  {number} n the number to process.
110      * @return {string}   the number with the leading zero added (if necessary).
111      * @private
112      */
113     function leadingZero(n) {
114         return (n < 10 ? '0' : '') + n;
115     }
116 
117     /**
118      * Returns the current time along with a formatted string in the specified
119      * form (see silog.DT_FORMAT)
120      * @param  {Object.<string, number>} format the format in which the current
121      *                                   date/ time should be returned.
122      * @return {Array.<string, number>}  the formatted date/ time along with the
123      *                                   original timestamp.
124      * @private
125      */
126     function getFormattedTimestamp(format) {
127         var date = new Date();
128         switch (format) {
129         case DT_FORMAT.DATE_TIME:
130             return [[date.getFullYear(), '/',
131                     leadingZero(date.getMonth()), '/',
132                     leadingZero(date.getDate()), ' ',
133                     leadingZero(date.getHours()), ':',
134                     leadingZero(date.getMinutes()), ':',
135                     leadingZero(date.getSeconds())].join(''), date];
136         case DT_FORMAT.DATE:
137             return [[date.getFullYear(), '/',
138                      leadingZero(date.getMonth()), '/',
139                      leadingZero(date.getDate())].join(''), date];
140         case DT_FORMAT.TIME:
141             return [[leadingZero(date.getHours()), ':',
142                      leadingZero(date.getMinutes()), ':',
143                      leadingZero(date.getSeconds())].join(''), date];
144         default:
145             // if the format is not in DT_FORMAT we use the datetime
146             return [[date.getFullYear(), '/',
147                     leadingZero(date.getMonth()), '/',
148                     leadingZero(date.getDate()), ' ',
149                     leadingZero(date.getHours()), ':',
150                     leadingZero(date.getMinutes()), ':',
151                     leadingZero(date.getSeconds())].join(''), date];
152         }
153     }
154 
155     /**
156      * Simple logger that outputs the messages to stdout.
157      * @param  {string}             what  the message to log.
158      * @param  {Object.<string, *>} extra extra data sent by silog.log().
159      */
160     function consoleLogger(what, extra) {
161         if (extra.hasOwnProperty('object')) {
162             what += ' - ' + JSON.stringify(extra['object']);
163         }
164         console.log(what);
165     }
166 
167     /**
168      * Constructor of the main class responsible for the logging logic.
169      *
170      * @param {Object} p configuration parameters, as follows (if not specified
171      *                   by user, defaults will be used):
172      *                   level: one of the elements of silog().LEVEL; only
173      *                          messages above it will be logged; default =
174      *                          LEVEL.INFO
175      *                   tsFormat: one of the elements of silog().DT_FORMAT;
176      *                             specifies how the date and time should be
177      *                             logged; default = DT_FORMAT.DATE_TIME
178      *                   loggers: array of functions that should be used for
179      *                            writing the logged messages to output; the
180      *                            signature of these function should be
181      *                            function write(message)
182      *                            where message is the message to be written; by
183      *                            default, console.log is used.
184      * @return {?Object} an instance of the Logger if the instantiation was
185      *                   successful, null otherwise.
186      * @this {Object} the Logger instance.
187      * @constructor
188      */
189     function Logger(p) {
190 
191         // default settings
192         this.level = LEVEL.INFO;
193         this.tsFormat = DT_FORMAT.DATE_TIME;
194         this.loggers = [consoleLogger];
195 
196         if (p.hasOwnProperty('level')) {
197             if (!checkLevel(p.level, true)) {
198                 this.wtf(LOCAL_TAG, 'Invalid message level: ' + p.level);
199                 return null;
200             } else {
201                 this.level = p.level;
202             }
203         }
204 
205         if (p.hasOwnProperty('tsformat')) {
206             for (var key in DT_FORMAT) {
207                 if (DT_FORMAT.hasOwnProperty(key)) {
208                     if (p.tsformat === DT_FORMAT[key]) {
209                         this.tsFormat = p.tsformat;
210                         break;
211                     }
212                 }
213             }
214 
215             if (typeof this.tsFormat === 'undefined') {
216                 this.wtf(LOCAL_TAG, 'Invalid date/ time format: ' + p.tsformat);
217                 return null;
218             }
219         }
220 
221         if (p.hasOwnProperty('loggers')) {
222             if (Array.isArray(p.loggers)) {
223                 for (var i = 0, len = p.loggers.length; i < len; i++) {
224                     if (typeof p.loggers[i] !== 'function') {
225                         this.wtf(LOCAL_TAG,
226                            'Invalid loggers: should be an array of functions');
227                         return null;
228                     }
229                 }
230                 this.loggers = p.loggers;
231             } else {
232                 this.wtf(LOCAL_TAG, 'Invalid loggers: should be an array');
233                 return null;
234             }
235         }
236 
237     }
238 
239 
240     /**
241      * Send a given messages to the various loggers.
242      * @param {Array.<number, string>} messageLevel the message logging level.
243      * @param {string} tag             the tag of the message.
244      * @param {string} message         the message to log.
245      * @param {*} object               an extra message to log.
246      * @this {Object} the Logger instance.
247      * @public
248      * @memberOf silog-Logger
249      */
250     Logger.prototype.log = function(messageLevel, tag, message, object) {
251         if (!checkLevel(messageLevel, false)) {
252             // we do not check if the level exists in silog.LEVEL as this can
253             // become too costly
254             this.w(LOCAL_TAG, ['Invalid message level for: ',
255                                 tag,
256                                 ' - ',
257                                 message,
258                                 ' = ',
259                                 messageLevel].join(''));
260         }
261 
262         if (messageLevel[0] >= this.level[0]) {
263             var time = getFormattedTimestamp(this.tsFormat);
264             var what = ['[',
265                         messageLevel[1],
266                         '] - ',
267                         time[0],
268                         ' - ',
269                         tag,
270                         ' - ',
271                         message].join('');
272 
273             // the extra information we send to the loggers
274             var extra = {'tag': tag,
275                          'message': message,
276                          'level': messageLevel,
277                          'ts': time[1] };
278 
279             if (object) {
280                 extra['object'] = object;
281             }
282 
283             for (var i = 0, len = this.loggers.length; i < len; i += 1) {
284                 this.loggers[i](what, extra);
285             }
286         } // end if (messageLevel[0] >= this.level[0])
287     };

290     /**
291      * Logs a message with the silog.LEVEL.ASSERT level.
292      * @param {string} tag     the tag of the message.
293      * @param {string} message the message to log.
294      * @param {*} object       an extra message to log.
295      * @this {Object} the Logger instance.
296      * @public
297      * @memberOf silog-Logger
298      */
299     Logger.prototype.wtf = function(tag, message, object) {
300         this.log(LEVEL.ASSERT, tag, message, object);
301     };
302 
303     /**
304      * Logs a message with the silog.LEVEL.ERROR level.
305      * @param {string} tag     the tag of the message.
306      * @param {string} message the message to log.
307      * @param {*} object       an extra message to log.
308      * @this {Object} the Logger instance.
309      * @public
310      * @memberOf silog-Logger
311      */
312     Logger.prototype.e = function(tag, message, object) {
313         this.log(LEVEL.ERROR, tag, message, object);
314     };
315

317      * Logs a message with the silog.LEVEL.WARN level.
318      * @param {string} tag     the tag of the message.
319      * @param {string} message the message to log.
320      * @param {*} object       an extra message to log.
321      * @this {Object} the Logger instance.
322      * @public
323      * @memberOf silog-Logger
324      */
325     Logger.prototype.w = function(tag, message, object) {
326         this.log(LEVEL.WARN, tag, message, object);
327     };

330      * Logs a message with the silog.LEVEL.INFO level.
331      * @param {string} tag     the tag of the message.
332      * @param {string} message the message to log.
333      * @param {*} object       an extra message to log.
334      * @this {Object} the Logger instance.
335      * @public
336      * @memberOf silog-Logger
337      */
338     Logger.prototype.i = function(tag, message, object) {
339         this.log(LEVEL.INFO, tag, message, object);
340     };
341 
342     /**
343      * Logs a message with the silog.LEVEL.DEBUG level.
344      * @param {string} tag     the tag of the message.
345      * @param {string} message the message to log.
346      * @param {*} object       an extra message to log.
347      * @this {Object} the Logger instance.
348      * @public
349      * @memberOf silog-Logger
350      */
351     Logger.prototype.d = function(tag, message, object) {
352         this.log(LEVEL.DEBUG, tag, message, object);
353     };
354 
355     /**
356      * Logs a message with the silog.LEVEL.VERBOSE level.
357      * @param {string} tag     the tag of the message.
358      * @param {string} message the message to log.
359      * @param {*} object       an extra message to log.
360      * @this {Object} the Logger instance.
361      * @public
362      * @memberOf silog-Logger
363      */
364     Logger.prototype.v = function(tag, message, object) {
365         this.log(LEVEL.VERBOSE, tag, message, object);
366     };

368 
369     // export public members
370     return {
371         level: LEVEL,
372         tsFormat: DT_FORMAT,
373         Logger: Logger,
374         consoleLogger: consoleLogger
375     };
376 };
